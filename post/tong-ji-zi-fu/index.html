统计字符

2019-08-08 1 min read 
#include <stdio.h>
int main(){
    int letter,blank,digit,other;
    char ch;
    int i;
    digit=other=blank=letter=0;
    for(i=1;i<=10;i++){
        ch=getchar();
        if((ch>='a'&&ch<='z')||(ch>='A'&&ch<='Z')){
            letter++;
        }
        else if(ch>='0'&&ch<='9'){
            digit++;
        }
        else if(ch==' '||ch=='\n'){
            blank++;
        }
        else
            other++;
    }
    printf("letter = %d, blank = %d, digit = %d, other = %d",letter,blank,digit,other);
    return 0;
}

注意英文字符包括大写字母
